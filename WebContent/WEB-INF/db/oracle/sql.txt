[Oracle Tablespace and User Create]
-- 테이블 스페이스 생성
CREATE TABLESPACE TS_MARUARA
DATAFILE 'C:\app\Administrator\oradata\orcl\MARUARA01.DBF'
SIZE 100M
AUTOEXTEND ON NEXT 10M
EXTENT MANAGEMENT LOCAL
UNIFORM SIZE 1M

-- 템프 테이블 스페이스 생성
CREATE TEMPORARY TABLESPACE TS_TEMP
TEMPFILE 'C:\app\Administrator\oradata\orcl\MARUARA_TEMP01.DBF'
SIZE 10M
EXTENT MANAGEMENT LOCAL
UNIFORM SIZE 512K

-- 유저 생성
CREATE USER MARUARA
IDENTIFIED BY MARUARA
DEFAULT TABLESPACE TS_MARUARA
TEMPORARY TABLESPACE TS_TEMP

-- 유저 권한 부여
GRANT RESOURCE, CONNECT TO MARUARA

CREATE TABLE TB_USER (
	ID VARCHAR2(100) NOT NULL,
	PW VARCHAR2(100) NOT NULL,
	NAME VARCHAR2(50) NOT NULL,
	ADDR VARCHAR2(4000),
	PHONE NUMBER,
	REGDATE DATE,
	MODDATE TIMESTAMP,
	MEMO CLOB,
	CONSTRAINT TB_USER_ID_PK PRIMARY KEY (ID)
)
